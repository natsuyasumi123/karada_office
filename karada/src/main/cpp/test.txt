void MyGLRenderContext::getKaradaData(float* kData , int index , int type){
	switch (type) {
		case 0 :  //koshi
            float koshiPosition = 0.7f ;
			kData[0 * 3] = karadaData[index][12 *3 ] ;
			kData[0 * 3 + 1 ] = karadaData[index][12 *3 + 1 ] ;

            kData[1 * 3] = karadaData[index][23 *3 ] /2 + karadaData[index][24 *3 ] /2 ;
            kData[1 * 3 + 1 ] = karadaData[index][23 *3 + 1 ] /2 + karadaData[index][24 *3 + 1 ] /2;

            kData[2 * 3] = karadaData[index][11 *3 ] ;
			kData[2 * 3 + 1 ] = karadaData[index][11 *3 + 1 ] ;

            kData[3 *3 ] = karadaData[index][24 *3 ] * koshiPosition + karadaData[index][12 * 3]* (1.0f - koshiPosition);
            kData[3 *3 + 1] = karadaData[index][24 *3 +1 ] * koshiPosition + karadaData[index][12 * 3 +1 ] *(1.0f - koshiPosition) ;

            kData[4 *3 ] = karadaData[index][23 *3 ] * koshiPosition + karadaData[index][11 * 3]* (1.0f - koshiPosition);
            kData[4 *3 + 1] = karadaData[index][23 *3 +1 ] * koshiPosition + karadaData[index][11 * 3 +1 ] *(1.0f - koshiPosition) ;
			break ;
	}
}
